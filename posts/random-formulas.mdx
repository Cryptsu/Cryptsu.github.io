#include <iostream>

// Define a class
class MyClass {
public:
    // Constructor
    MyClass(int x, int y) : x_(x), y_(y) {}

    // Getter and setter functions
    int getX() const { return x_; }
    int getY() const { return y_; }
    void setX(int x) { x_ = x; }
    void setY(int y) { y_ = y; }

private:
    int x_;
    int y_;
};

int main() {
    // Create an instance of the class
    MyClass myObj(1, 2);

    // Use the getter and setter functions
    cout << "X: " << myObj.getX() << endl;
    myObj.setY(3);
    cout << "Y: " << myObj.getY() << endl;

    return 0;
}

package mail

import (

  "log"
  "net/smtp"

  "github.com/drone/drone/pkg/database"
  "github.com/drone/drone/pkg/template"
)

// A Message represents an email message. Addresses may be of any
// form permitted by RFC 822.
type Message struct {
  Sender   string
  ReplyTo  string // may be empty

  To       string
  Subject  string
  Body     string
}

// Sends a activation email to the User
func SendActivation(to string, data interface{}) error {
  msg := Message{}
  msg.Subject = "[drone.io] Account Activation"
  msg.To = to

  var buf bytes.Buffer
  err := template.ExecuteTemplate(&buf, "activation.html", &data)
  if err != nil {
    log.Print(err)
    return err
  }
  msg.Body = buf.String()

  return Send(&msg)
}
